- 安装依赖
```
yarn install
```

- 开发模式运行
```
yarn run dev
```

- 项目编译
```
yarn run build
```


- 项目目录

```
├── public
|   └── index.html           # Vue 入口模板
├── src
│   ├── api                  # Api
│   ├── assets               # 本地静态资源
│   ├── components           # 业务通用组件
│   ├── core                 # 项目引导, 全局配置初始化，依赖包引入等
│   ├── router               # Vue-Router
│   ├── store                # Vuex
│   ├── utils                # 工具库
│   ├── views                # 业务页面入口和常用模板
│   ├── App.vue              # Vue 模板入口
│   └── main.js              # Vue 入口 JS
│   └── global.less          # 全局样式
├── README.md
└── package.json
```
